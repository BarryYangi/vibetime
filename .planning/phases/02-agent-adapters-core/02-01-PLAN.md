    ```typescript
    import type { AdapterFn, EventType, NormalizedEvent } from '../events.js'

    // Per-vendor event-name → event_type lookup. Claude Code has all four.
    // (Other Claude events like UserPromptExpansion / PostToolBatch / etc. are
    // NOT in V0 scope per RESEARCH §A.1 drift notes — adapter returns null.)
    const EVENT_TYPES: Readonly<Record<string, EventType>> = {
      UserPromptSubmit: 'turn_start',
      Stop: 'turn_end',
      SessionStart: 'session_start',
      SessionEnd: 'session_end',
    }

    // Per-vendor required-fields type guard.
    function hasRequired(p: unknown): p is { session_id: string; cwd: string } {
      return (
        p !== null &&
        typeof p === 'object' &&
        typeof (p as { session_id?: unknown }).session_id === 'string' &&
        typeof (p as { cwd?: unknown }).cwd === 'string'
      )
    }

    /**
     * Claude Code adapter (PRD §8 / DEC-011 / ADPT-01).
     *
     * Maps `UserPromptSubmit` / `Stop` / `SessionStart` / `SessionEnd` to a
     * NormalizedEvent. `turn_id` is derived as `${session_id}-${ts}` (PRD §8
     * — Claude has no native turn ID).
     *
     * Returns a NormalizedEvent whose `project` field is the **raw cwd string**
     * extracted from the vendor payload. The hook layer (Phase 3) is responsible
     * for replacing this with the result of `resolveProject({ cwd, aliases,
     * gitRemoteUrl })` BEFORE persisting to SQLite. See PRD §6 + DEC-010 +
     * RESEARCH §D Option 3.
     *
     * Pure. Never throws. Returns null on unparseable input.
     */
    export const adaptClaudeCode: AdapterFn = (rawPayload, eventName) => {
      // Belt-and-braces: explicit type guards below already cover documented
      // inputs; this catch is for exotic Proxy/getter abuse to satisfy the
      // never-throws contract (DEC-011). Same intentional pattern as
      // resolveProject in project.ts:62-97. NOT redundant.
      try {
        const event_type = EVENT_TYPES[eventName]
        if (!event_type) return null

        if (!hasRequired(rawPayload)) return null

        // ts at call time — NOT module scope (RESEARCH §G.1).
        const ts = Date.now() / 1000
        const timezone = Intl.DateTimeFormat().resolvedOptions().timeZone

        // turn_id derivation per PRD §8: only for turn_* events.
        // Use unrounded ts to dodge same-second collisions (RESEARCH §G.7).
        const turn_id =
          event_type === 'turn_start' || event_type === 'turn_end'
            ? `${rawPayload.session_id}-${ts}`
            : undefined

        // V0 meta whitelist: model + source on session_start; reason on session_end.
        // (Drop prompt/stop_reason/permission_mode etc. per RESEARCH §G.8.)
        const meta: Record<string, unknown> = {}
        if (event_type === 'session_start') {
          const model = (rawPayload as { model?: unknown }).model
          if (typeof model === 'string') meta.model = model
          const source = (rawPayload as { source?: unknown }).source
          if (typeof source === 'string') meta.source = source
        }
        if (event_type === 'session_end') {
          const reason = (rawPayload as { reason?: unknown }).reason
          if (typeof reason === 'string') meta.reason = reason
        }

        const event: NormalizedEvent = {
          agent: 'claude-code',
          event_type,
          project: rawPayload.cwd,
          session_id: rawPayload.session_id,
          ts,
          timezone,
          // OMIT (not undefined) when not present — exactOptionalPropertyTypes:
          ...(turn_id !== undefined ? { turn_id } : {}),
          ...(Object.keys(meta).length > 0 ? { meta } : {}),
        }
        return event
      } catch {
        // See try-block comment above. Last line of defense.
        return null
      }
    }
    ```

     * Pure. Never throws.
     */

      try {
        const event_type = EVENT_TYPES[eventName]
        if (!event_type) return null

        const isTurn = event_type === 'turn_start' || event_type === 'turn_end'

        if (isTurn) {
          if (!hasRequiredTurn(rawPayload)) return null
        } else {
          if (!hasRequiredSession(rawPayload)) return null
        }

        const timezone = Intl.DateTimeFormat().resolvedOptions().timeZone

        // workspace_roots[0] extraction with empty/missing guard (RESEARCH §G.4).
        // noUncheckedIndexedAccess makes ws[0] return string | undefined.

        // session_id ← conversation_id; turn_id ← generation_id (turn events only).
        const session_id = (rawPayload as { conversation_id: string }).conversation_id
        const turn_id = isTurn ? (rawPayload as { generation_id: string }).generation_id : undefined

        // V0 meta whitelist:
        const meta: Record<string, unknown> = {}
        if (event_type === 'session_start') {
          // RESEARCH §A.3 doc-asymmetry: model may or may not be present.
          const model = (rawPayload as { model?: unknown }).model
          if (typeof model === 'string') meta.model = model
        }
        if (event_type === 'session_end') {
          const reason = (rawPayload as { reason?: unknown }).reason
          if (typeof reason === 'string') meta.reason = reason
        }

        const event: NormalizedEvent = {
          agent: 'cursor',
          event_type,
          project: cwd,
          session_id,
          ts,
          timezone,
          ...(turn_id !== undefined ? { turn_id } : {}),
          ...(Object.keys(meta).length > 0 ? { meta } : {}),
        }
        return event
      } catch {
        return null
      }
    }

    // Reference: https://stackoverflow.com/a/47593316 (public domain).
    function mulberry32(seed: number): () => number {
      let a = seed >>> 0
      return () => {
        a = (a + 0x6d2b79f5) >>> 0
        let t = a
        t = Math.imul(t ^ (t >>> 15), t | 1)
        t ^= t + Math.imul(t ^ (t >>> 7), t | 61)
        return ((t ^ (t >>> 14)) >>> 0) / 4294967296
      }

      if (r < 0.38) return 42
      if (r < 0.42) return []

      if (rng() < 0.4) {
